def all
  @out_file = Path.get_out_file_txt(@name)
  cmd = @path + " -m -n -i any |tee " + @out_file
  run(cmd)
  services = Dir.pwd + "/dsniff.services"
  if File.exists?(services)
    FileUtils.rm(services)
  end
  clean_exit
end
