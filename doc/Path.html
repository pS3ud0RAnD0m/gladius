def req
  require "fileutils"
  require "pty"
  require "time"
  require @weapons + "weapon"
  Dir[@menus + "*.rb"].each { |a| require a }
  Dir[@weapons + "*.rb"].each { |a| require a }
  Dir[@helpers + "*.rb"].each { |a| require a }
  Dir[@thirdparty + "*.rb"].each { |a| require a }
end
